368 template <class T>
369 void enqueue_mail(Mail<T, 16> &mail_queue, T val) {
370  auto mail_item = mail_queue.alloc();
371  *mail_item = val;
372  if(!mail_queue.full()) {
373  mail_queue.put(mail_item);
374  }
375  else {
376  debug_printf(DBG_WARN, "Mail queue is full!\r\n");
377  }
378 }
379 
385 template <class T>
386 T dequeue_mail(Mail<T, 16> &mail_queue) {
387  T mail_item;
388  for(;;) {
389  osEvent evt = mail_queue.get();
390  if(evt.status == osEventMail) {
391  mail_item = *((T *) evt.value.p);
392  mail_queue.free((T *) evt.value.p);
393  break;
394  }
395  else { MBED_ASSERT(false); }
396  }
397  return mail_item;
398 }
